simple config for nginx /etc/nginx/sites-availables/default

server {
	listen 80 default_server;
	listen [::]:80 default_server;


	 listen 443 ssl default_server;
	 listen [::]:443 ssl default_server;
	 ssl_certificate /etc/letsencrypt/live/rozkladnpubot.tk/fullchain.pem;
	 ssl_certificate_key /etc/letsencrypt/live/rozkladnpubot.tk/privkey.pem;

	root /var/www/html;

	index index.html index.htm index.nginx-debian.html;

	server_name _;

	location / {
		
proxy_pass http://localhost:1616;		
	}
	
}
